2. ROS 2节点


class HttpBridgeNode(Node):
    def __init__(self):
        super().__init__('http_bridge_node')
        
        # 订阅: 接收硬件数据
        self.can_rx_sub = self.create_subscription(
            String, '/can_rx', 
            self.can_rx_callback, 10
        )
        
        # 发布: 发送控制命令
        self.can_tx_pub = self.create_publisher(
            String, '/can_tx', 10
        )
    
    def can_rx_callback(self, msg):
        """收到CAN数据时更新全局状态"""
        data = json.loads(msg.data)
        with state.lock:
            state.motor1_angle = data.get('motor1_angle')

4. 主函数 - 启动两个线程


def main():
    # 初始化ROS
    rclpy.init()
    global ros_node
    ros_node = HttpBridgeNode()
    
    # 线程1: Flask (daemon=True表示后台线程)
    flask_thread = Thread(
        target=lambda: app.run(host='0.0.0.0', port=8081),
        daemon=True
    )
    flask_thread.start()
    
    # 线程2: ROS (主线程)
    rclpy.spin(ros_node)  # 阻塞，持续处理ROS消息
为什么Flask是daemon线程?


daemon=True  # 主线程结束时，daemon线程自动结束
daemon=False # 主线程结束时，等待该线程结束

# 我们的情况:
# 主线程 = ROS spin (Ctrl+C停止)
# Flask线程 = daemon
# 按Ctrl+C → ROS停止 → Flask自动停止 ✅

C++代码结构

class CameraWebSocketClient {
private:
    net::io_context ioc_;                    // Boost.Asio IO上下文
    tcp::resolver resolver_;                 // DNS解析器
    websocket::stream<tcp::socket> ws_;      // WebSocket流
    cv::VideoCapture camera_;                // OpenCV摄像头
    
public:
    void connect() {
        // 1. DNS解析
        auto results = resolver_.resolve(server_host_, server_port_);
        
        // 2. TCP连接
        net::connect(ws_.next_layer(), results);
        
        // 3. WebSocket握手
        ws_.handshake(server_host_, "/");
    }
    
    void sendImage(const cv::Mat& frame) {
        // 1. JPEG压缩
        std::vector<uchar> buffer;
        cv::imencode(".jpg", frame, buffer);
        
        // 2. Base64编码
        std::string base64 = base64_encode(buffer);
        
        // 3. JSON封装
        json j = {
            {"type", "image"},
            {"data", base64}
        };
        
        // 4. WebSocket发送
        ws_.write(net::buffer(j.dump()));
    }
};
为什么用WebSocket而不是HTTP?
特性	HTTP	WebSocket
连接	每次请求建立	持久连接
开销	每帧都有HTTP头	只有首次握手
延迟	高	低
适用	偶尔请求	实时流
视频流场景:


HTTP方式 (每秒10帧):
连接 → 发送帧1 → 断开
连接 → 发送帧2 → 断开
...
= 10次TCP握手/秒 ❌

WebSocket方式:
连接 (一次)
发送帧1
发送帧2
...
= 1次TCP握手 ✅

4. 多线程同步
为什么需要Lock?


# 没有锁的情况:
# 线程1                    线程2
angle = state.motor1_angle  # 读取: 0.0
                            state.motor1_angle = 45.0  # 写入

# 线程1                    线程2
with state.lock:            # 等待...

print(angle)  # 输出: 0.0 (正确!)
Lock的实现:


from threading import Lock

lock = Lock()

# 方式1: 手动
lock.acquire()
try:
    # 临界区代码
    state.motor1_angle = 45.0
finally:
    lock.release()

# 方式2: with语句 (推荐)
with lock:
    # 临界区代码
    state.motor1_angle = 45.0
# 自动释放锁

#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from flask import Flask, request, jsonify
from threading import Thread, Lock
import json

# ========== 1. 全局状态 ==========
class SystemState:
    def __init__(self):
        self.lock = Lock()
        self.mode = "active"
        self.motor1_angle = 0.0
        # ...

state = SystemState()

# ========== 2. ROS 2节点 ==========
class HttpBridgeNode(Node):
    def __init__(self):
        super().__init__('http_bridge_node')
        
        # 订阅CAN数据
        self.can_rx_sub = self.create_subscription(
            String, '/can_rx', 
            self.can_rx_callback, 10
        )
        
        # 发布控制命令
        self.can_tx_pub = self.create_publisher(
            String, '/can_tx', 10
        )
    
    def can_rx_callback(self, msg):
        """收到CAN数据时更新状态"""
        data = json.loads(msg.data)
        with state.lock:
            state.motor1_angle = data.get('motor1_angle')
    
    def publish_command(self, cmd_dict):
        """发布命令到CAN"""
        msg = String()
        msg.data = json.dumps(cmd_dict)
        self.can_tx_pub.publish(msg)

        "error": None
    })

    global ros_node

    rclpy.init()
    ros_node = HttpBridgeNode()

    flask_thread = Thread(

        daemon=True
    )
    flask_thread.start()

    try:
        rclpy.spin(ros_node)
    except KeyboardInterrupt:
        pass
    finally:
        ros_node.destroy_node()
        rclpy.shutdown()
